<h2>样式

创建文件components/content/ProseH2.vue，拷贝ProseH2.vue中的内容，修改后文件内容如下：

components/content/ProseH2.vue
<template>
    <h2 :id="id" class="mt-8 mb-4">
        <a v-if="id && generate" :href="`#${id}`" class="ml-n12 mr-4">
            <v-icon icon="mdi-format-header-2" size="small" />
        </a>
        <slot />
    </h2>
</template>

<script setup lang="ts">
import { useRuntimeConfig } from '#imports'
defineProps<{ id?: string }>()
const heading = 2
const { anchorLinks } = useRuntimeConfig().public.content
const generate = anchorLinks?.depth >= heading && !anchorLinks?.exclude.includes(heading)
</script>

加亮代码涉及的技巧：

  • class="m*-* p*-*"等用于调整元素间距。阅读间距 — Vuetify。
  • v-icon用于添加图标。阅读图标字体 — Vuetify。

为了使用mdi-format-header-2图标，需要执行下列操作：

  • 安装Material Design图标
terminal
npm install @mdi/font --save-dev
  • 修改plugins/vuetify.js
plugins/vuetify.js
import { createVuetify } from 'vuetify'
import * as components from 'vuetify/components'
import * as directives from 'vuetify/directives'
import '@mdi/font/css/materialdesignicons.css'

export default defineNuxtPlugin(nuxtApp => {
  const vuetify = createVuetify({
    components,
    directives,
    ssr: true
  })

  nuxtApp.vueApp.use(vuetify)
})

<code>样式

创建文件components/content/ProseCode.vue，拷贝ProseCode.vue中的内容，修改后文件内容如下：

components/content/ProseCode.vue
<template>
    <v-hover v-slot="{ isHovering, props }" close-delay="100">
        <v-card :elevation="isHovering ? 4 : 0" v-bind="props" variant="outlined">
            <Transition>
                <v-chip v-if="!isHovering" variant="text" :text="filename"
                    style="position: absolute; right: 0; top: 0; font-family: consolas;" />
            </Transition>

            <v-container class="overflow-x-auto pa-0">
                <v-card-text style="display: inline-block; min-width: 100%;">
                    <slot />
                </v-card-text>
            </v-container>

            <Transition>
                <v-card-actions v-if="isHovering" style="position: absolute; right: 0; bottom: 0">
                    <v-btn icon="mdi-content-copy" size="small" @click="copyCode()" variant="plain" />
                </v-card-actions>
            </Transition>
        </v-card>
    </v-hover>
</template>

<script lang="ts">
import { defineComponent } from '#imports'

export default defineComponent({
    props: {
        code: {
            type: String,
            default: ''
        },
        language: {
            type: String,
            default: null
        },
        filename: {
            type: String,
            default: null
        },
        highlights: {
            type: Array as () => number[],
            default: () => []
        },
        meta: {
            type: String,
            default: null
        }
    },
    methods: {
        copyCode() {
            var promise = navigator.clipboard.writeText(this.code);
        }
    }
})
</script>

<style>
pre code .line {
    display: block;
    min-height: 1rem;
    font-family: consolas;
}

.highlight {
    background-color: greenyellow;
}

.v-enter-active,
.v-leave-active {
    transition: opacity 0.5s ease;
}

.v-enter-from,
.v-leave-to {
    opacity: 0;
}
</style>

加亮代码涉及的技巧：

  • Transiton和v-if搭配使用实现组件平缓出现和隐藏。阅读Transition | Vue.js。
  • style="position: absolute; right: 0; top: 0; font-family: consolas;"css属性，实现组件浮动在另一个组件之上。阅读position - CSS：层叠样式表 | MDN。
  • class="overflow-x-auto"实现组件在超过屏幕宽度是可以滑动的。阅读Overflow — Vuetify。
  • @click=""是vue指令，用于监听点击事件，注意不要写成onclick。阅读事件处理 | Vue.js
  • navigator.clipboard.writeText(this.code);写入剪切板函数接口。阅读Clipboard.writeText() - Web API 接口参考 | MDN。

<img>样式

创建文件components/content/ProseImg.vue，拷贝ProseImg.vue中的内容，修改后文件内容如下：

components/content/ProseImg.vue
<template>
    <v-hover v-slot="{ isHovering, props }" close-delay="100">
        <v-card :elevation="isHovering ? 4 : 0" v-bind="props" variant="outlined" :ripple="false">
            <v-img :src="refinedSrc" :alt="alt"></v-img>
        </v-card>

        <v-overlay activator="parent" scroll-strategy="block" close-on-back close-on-content-click
            class="align-center justify-center">
            <v-img :src="refinedSrc" :alt="alt"></v-img>
        </v-overlay>
    </v-hover>
</template>
  
<script setup lang="ts">
import { withBase } from 'ufo'
import { useRuntimeConfig, computed } from '#imports'

const props = defineProps({
    src: {
        type: String,
        default: ''
    },
    alt: {
        type: String,
        default: ''
    },
    width: {
        type: [String, Number],
        default: undefined
    },
    height: {
        type: [String, Number],
        default: undefined
    }
})

const refinedSrc = computed(() => {
    if (props.src?.startsWith('/') && !props.src.startsWith('//')) {
        return withBase(props.src, useRuntimeConfig().app.baseURL)
    }
    return props.src
})
</script>

加亮代码涉及的技巧：

  • v-overlay组件。阅读Overlay 遮罩层 — Vuetify。
  • class="align-center justify-center"用于元素排布。阅读弹性布局 — Vuetify。
